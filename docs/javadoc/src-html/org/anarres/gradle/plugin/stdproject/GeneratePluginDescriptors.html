006package org.anarres.gradle.plugin.stdproject;
007
008import java.io.File;
009import java.util.Map;
010import javax.annotation.Nonnull;
011import org.codehaus.groovy.runtime.DefaultGroovyMethods;
012import org.gradle.api.internal.ConventionTask;
013import org.gradle.api.tasks.Input;
014import org.gradle.api.tasks.OutputDirectory;
015import org.gradle.api.tasks.TaskAction;
016
017/**
018 *
019 * @author shevek
020 */
021public class GeneratePluginDescriptors extends ConventionTask {
022
023    private File destinationDir;
024    private Map<String, String> pluginImplementations;
025
026    @OutputDirectory
027    public File getDestinationDir() {
028        return destinationDir;
029    }
030
031    public void setDestinationDir(@Nonnull File destinationDir) {
032        this.destinationDir = destinationDir;
033    }
034
035    @Input
036    public Map<String, String> getPluginImplementations() {
037        return pluginImplementations;
038    }
039
040    public void setPluginImplementations(@Nonnull Map<String, String> pluginImplementations) {
041        this.pluginImplementations = pluginImplementations;
042    }
043
044    @TaskAction
045    public void generate() throws Exception {
046        final File destinationDir = getDestinationDir();
047        DefaultGroovyMethods.deleteDir(destinationDir);
048        for (Map.Entry<String, ? extends Object> e : getPluginImplementations().entrySet()) {
049            File pluginDescriptorFile = new File(destinationDir, "META-INF/gradle-plugins/" + e.getKey() + ".properties");
050            pluginDescriptorFile.getParentFile().mkdirs();
051            DefaultGroovyMethods.setText(pluginDescriptorFile,
052                    "implementation-class=" + e.getValue() + "\n"
053                    + "implementation-version=" + getProject().getVersion() + "\n");
054        }
055    }
056}
