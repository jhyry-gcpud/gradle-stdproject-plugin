001package org.anarres.gradle.plugin.stdproject;
002
003import com.google.common.base.Objects;
004import groovy.lang.Closure;
005import groovy.lang.DelegatesTo;
006import groovy.lang.GroovyObjectSupport;
007import java.util.ArrayList;
008import java.util.List;
009import java.util.Map;
010import javax.annotation.Nonnull;
011import org.codehaus.groovy.runtime.DefaultGroovyMethods;
012import org.gradle.api.Project;
013import org.gradle.util.ConfigureUtil;
014import org.slf4j.Logger;
015import org.slf4j.LoggerFactory;
016
017/**
018 * The standard module plugin extension.
019 *
020 * This allows configuring the standard module plugin using a
021 * <code>stdmodule { }</code> block.
022 *
023 * @author shevek
024 */
025public class StdModuleExtension extends GroovyObjectSupport {
026
027    private static final Logger LOG = LoggerFactory.getLogger(StdModuleExtension.class);
028    private static final String GITHUB_URL_PREFIX = "https://github.com/";
029
030    public static class Person extends GroovyObjectSupport {
031
032        public String id;
033        public String name;
034        public String email;
035
036        public void id(@Nonnull String id) {
037            this.id = id;
038        }
039
040        public void name(@Nonnull String name) {
041            this.name = name;
042        }
043
044        public void email(@Nonnull String email) {
045            this.email = email;
046        }
047
048        @Override
049        public String toString() {
050            return Objects.toStringHelper(this)
051                    .add("id", id)
052                    .add("name", name)
053                    .add("email", email)
054                    .toString();
055        }
056    }
057
058    public String description;
059    public final String url;
060    public final String issuesSystem = "github";
061    public final String issuesUrl;
062    public final String vcsUrl;
063    public String inceptionYear;
064    public List<Person> authors = new ArrayList<Person>();
065    public List<License> licenses = new ArrayList<License>();
066
067    public StdModuleExtension(@Nonnull Project project) {
068        String githubPath = StdProjectPlugin.getGithubPath(project);
069        url = GITHUB_URL_PREFIX + githubPath;
070        issuesUrl = GITHUB_URL_PREFIX + githubPath + "/issues";
071        vcsUrl = "scm:git:git@github.com:" + githubPath + ".git";
072    }
073
074    public void description(@Nonnull String description) {
075        this.description = description;
076    }
077
078    public void author(@Nonnull Map<String, Object> m) {
079        Person person = new Person();
080        ConfigureUtil.configureByMap(m, person);
081        authors.add(person);
082    }
083
084    public void author(@Nonnull @DelegatesTo(Person.class) Closure c) {
085        Person person = new Person();
086        ConfigureUtil.configure(c, person);
087        // LOG.info("Adding author " + person);
088        authors.add(person);
089    }
090
091    public void license(@Nonnull String name) {
092        License license = License.LICENSES.get(name);
093        if (license == null)
094            throw new IllegalArgumentException("Unknown license " + name + "; available are " + License.LICENSES.keySet() + " or configure with a Closure.");
095        licenses.add(license);
096    }
097
098    public void license(@Nonnull Map<String, Object> m) {
099        License license = new License();
100        ConfigureUtil.configureByMap(m, license, DefaultGroovyMethods.getProperties(license).keySet());
101        licenses.add(license);
102    }
103
104    public void license(@Nonnull @DelegatesTo(License.class) Closure closure) {
105        License license = new License();
106        ConfigureUtil.configure(closure, license);
107        licenses.add(license);
108    }
109}
