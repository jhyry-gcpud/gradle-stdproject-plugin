038/**
039 * The standard module plugin.
040 *
041 * @author shevek
042 */
043public class StdModulePlugin implements Plugin<Project> {
044
045    private static final Logger LOG = LoggerFactory.getLogger(StdModulePlugin.class);
046
047    @Override
048    public void apply(final Project project) {
049        // Root
050        project.getRootProject().getPlugins().apply(StdProjectPlugin.class);
051
052        // Extensions
053        // final StdProjectExtension rootExtension = project.getRootProject().getExtensions().getByType(StdProjectExtension.class);
054        final StdModuleExtension extension = project.getExtensions().create("stdmodule", StdModuleExtension.class, project);
055
056        // Convention
057        project.getPlugins().apply(JavaPlugin.class);
058        project.getPlugins().apply(InfoPlugin.class);
059        project.getPlugins().apply(VersionsPlugin.class);
060
061        JavaPluginConvention java = project.getConvention().getPlugin(JavaPluginConvention.class);
062        java.setSourceCompatibility(JavaVersion.VERSION_1_7);
063
064        project.getRepositories().add(project.getRepositories().mavenCentral());
065        project.getRepositories().add(project.getRepositories().jcenter());
066
067        project.getDependencies().add("testCompile", "junit:junit:4.12");
068        project.getDependencies().add("testCompile", "org.slf4j:slf4j-api:1.7.12");
069        project.getDependencies().add("testRuntime", "ch.qos.logback:logback-classic:1.1.3");
070
071        // Javadoc javadoc = (Javadoc) project.getTasks().getByName(JavaPlugin.JAVADOC_TASK_NAME);
072        for (Javadoc javadoc : project.getTasks().withType(Javadoc.class))
073            StdTaskConfiguration.configureJavadoc(project, javadoc);
074
075        Test test = (Test) project.getTasks().getByName(JavaPlugin.TEST_TASK_NAME);
076        test.systemProperty("org.apache.commons.logging.Log", "org.apache.commons.logging.impl.SimpleLog");
077        test.systemProperty("org.apache.commons.logging.simplelog.defaultlog", "debug");
078        test.systemProperty("org.apache.commons.logging.diagnostics.dest", "STDERR");
079        test.systemProperty("org.slf4j.simpleLogger.logFile", "System.out");
080        test.systemProperty("org.slf4j.simpleLogger.defaultLogLevel", "debug");
081        test.systemProperty("jna.nosys", "true");
082        final String testSystemPropertyPrefix = "test-sys-prop.";
083        for (Map.Entry<Object, ? extends Object> e : System.getProperties().entrySet()) {
084            String key = String.valueOf(e.getKey());
085            if (key.startsWith(testSystemPropertyPrefix))
086                test.systemProperty(key.substring(testSystemPropertyPrefix.length()), e.getValue());
087        }
088        for (Map.Entry<String, ? extends Object> e : project.getExtensions().getExtraProperties().getProperties().entrySet()) {
089            String key = e.getKey();
090            if (key.startsWith(testSystemPropertyPrefix))
091                test.systemProperty(key.substring(testSystemPropertyPrefix.length()), e.getValue());
092        }
093
094        TestLoggingContainer testLogging = test.getTestLogging();
095        testLogging.events(TestLogEvent.STARTED, TestLogEvent.PASSED, TestLogEvent.FAILED, TestLogEvent.SKIPPED);
096        if (System.getProperty("test.single") != null) {
097            testLogging.setShowStackTraces(true);
098            testLogging.setShowExceptions(true);
099            testLogging.setExceptionFormat(TestExceptionFormat.FULL);
100            testLogging.setShowStandardStreams(true);
101        }
102
103        // Check
104        project.getPlugins().apply(FindBugsPlugin.class);
105        FindBugsExtension findbugs = project.getExtensions().getByType(FindBugsExtension.class);
106        findbugs.setIgnoreFailures(true);   // Hope the plugin set this property as a convention.
107        project.getTasks().getByName("findbugsTest").setEnabled(false);
108
109        project.getPlugins().apply(CoberturaPlugin.class);
110        CoberturaExtension cobertura = project.getExtensions().getByType(CoberturaExtension.class);
111        cobertura.setCoverageFormats(Sets.newHashSet("html", "xml"));
112        // https://github.com/stevesaliman/gradle-cobertura-plugin/issues/81
113        // cobertura.setCoberturaVersion("2.1.1");
114
115        project.getPlugins().apply(AnimalSnifferPlugin.class);
116        AnimalSnifferExtension animalSniffer = project.getExtensions().getByType(AnimalSnifferExtension.class);
117        animalSniffer.setSignature("org.codehaus.mojo.signature:java17:+@signature");
118
119        // Nexus
120        project.getPlugins().apply(NexusPlugin.class);
121        // final NexusPluginExtension nexus = project.getExtensions().getByType(NexusPluginExtension.class);
122        project.afterEvaluate(new Action<Project>() {
123            @Override
124            public void execute(Project t) {
125
126                for (Upload upload : project.getTasks().withType(Upload.class)) {
127                    for (MavenDeployer deployer : upload.getRepositories().withType(MavenDeployer.class))
128                        deployer.setUniqueVersion(false);
129
130                    for (MavenResolver resolver : upload.getRepositories().withType(MavenResolver.class)) {
131                        MavenPom pom = resolver.getPom();
132                        // We can't cast (Model) pom.getModel() because of FilteringClassLoader.
133
134                        /*
135                         t.getLogger().info("Task " + upload + " pom " + pom + " of " + pom.getClass());
136                         t.getLogger().info("Task " + upload + " model " + pom.getModel() + " of " + pom.getModel().getClass());
137                         CompositeQuery q = new CompositeQuery();
138                         q.add(new ClassQuery(pom.getClass()));
139                         q.add(new ClassQuery(pom.getModel().getClass()));
140                         q.add(new ClassQuery(getClass()));
141                         q.add(new ClassLoaderQuery("maven", pom.getModel().getClass().getClassLoader()));
142                         q.add(new ClassLoaderQuery("self", getClass().getClassLoader()));
143                         t.getLogger().info(String.valueOf(q.call()));
144                         */
145                        // developers is a List<Developer> and licenses is a List<License>
146                        // so they need special treatment.
147                        pom.project(new Closure(StdModulePlugin.this) {
148                            @Override
149                            public Object call(Object... args) {
150                                final Object pom = getDelegate();  // It's a CustomModelBuilder extends ModelBuilder extends FactoryBuilderSupport
151
152                                DefaultGroovyMethods.invokeMethod(pom, "licenses", new Closure(StdModulePlugin.this) {
153                                    // This delegates to the same ModelBuilder, with an internal state
154                                    // change to represent that we're now in the developers{} block.
155                                    // LOG.debug("Developers delegate is {}", getDelegate());
156                                    @Override
157                                    public Object call(Object... args) {
158                                        for (License license : extension.licenses) {
159                                            Object target = DefaultGroovyMethods.invokeMethod(pom, "license", null);
160                                            ConfigureUtil.configureByMap(ImmutableMap.<String, Object>of(
161                                                    "name", Preconditions.checkNotNull(license.getName(), "License name was null."),
162                                                    "url", Preconditions.checkNotNull(license.getUri(), "License URI was null."),
163                                                    "distribution", "repo"
164                                            ), target);
165                                        }
166                                        return null;
167                                    }
168                                });
169
170                                DefaultGroovyMethods.invokeMethod(pom, "developers", new Closure(StdModulePlugin.this) {
171                                    @Override
172                                    public Object call(Object... args) {
173                                        for (StdModuleExtension.Person person : extension.authors) {
174                                            Object target = DefaultGroovyMethods.invokeMethod(pom, "developer", null);
175                                            ConfigureUtil.configureByMap(ImmutableMap.<String, Object>of(
176                                                    "id", Preconditions.checkNotNull(person.id, "Person.id was null."),
177                                                    "name", Preconditions.checkNotNull(person.name, "Person.name was null."),
178                                                    "email", Preconditions.checkNotNull(person.email, "Person.email was null.")
179                                            ), target);
180                                            if (LOG.isDebugEnabled())
181                                                LOG.debug("Developer value is {} with props {}", target, DefaultGroovyMethods.getProperties(target));
182                                        }
183                                        return null;
184                                    }
185                                });
186
187                                return null;
188                            }
189                        });
190
191                        // Everything else is simple, and we can do this:
192                        Map<String, Object> pomData = new HashMap<String, Object>();
193                        pomData.put("name", project.getName());
194                        pomData.put("description", extension.description);
195                        pomData.put("url", extension.url);
196                        pomData.put("inceptionYear", extension.inceptionYear);
197                        pomData.put("scm", ImmutableMap.<String, Object>of(
198                                "connection", extension.vcsUrl,
199                                "url", extension.vcsUrl,
200                                "developerConnection", extension.vcsUrl
201                        ));
202                        pomData.put("issueManagement", ImmutableMap.<String, Object>of(
203                                "system", extension.issuesSystem,
204                                "url", extension.issuesUrl
205                        ));
206                        t.getLogger().info("Configuring " + pom + " with " + pomData);
207                        ConfigureUtil.configureByMap(pomData, pom.getModel());
208                    }
209                }
210
211            }
212        });
213    }
