006package org.anarres.gradle.plugin.stdproject;
007
008import java.io.File;
009import java.util.SortedSet;
010import java.util.TreeSet;
011import org.gradle.api.artifacts.ConfigurationContainer;
012import org.gradle.api.initialization.dsl.ScriptHandler;
013import org.gradle.api.internal.ConventionTask;
014import org.gradle.api.tasks.TaskAction;
015
016/**
017 *
018 * @author shevek
019 */
020public class BuildDependencies extends ConventionTask {
021
022    @TaskAction
023    public void run() {
024        SortedSet<String> dependencies = new TreeSet<String>();
025        ConfigurationContainer container = getProject().getBuildscript().getConfigurations();
026        for (File file : container.getByName(ScriptHandler.CLASSPATH_CONFIGURATION))
027            dependencies.add(file.getName());
028        for (String dependency : dependencies)
029            System.out.println(dependency);
030
031    }
032}
