006package org.anarres.gradle.plugin.stdproject;
007
008import javax.annotation.Nonnull;
009import org.gradle.api.Action;
010import org.gradle.api.JavaVersion;
011import org.gradle.api.Project;
012import org.gradle.api.Task;
013import org.gradle.api.tasks.javadoc.Javadoc;
014import org.gradle.external.javadoc.StandardJavadocDocletOptions;
015
016/**
017 *
018 * @author shevek
019 */
020public class StdTaskConfiguration {
021
022    public static void configureJavadoc(@Nonnull final Project project, @Nonnull final Javadoc javadoc) {
023        final StdProjectExtension extension = project.getRootProject().getExtensions().getByType(StdProjectExtension.class);
024        final StandardJavadocDocletOptions javadocOptions = (StandardJavadocDocletOptions) javadoc.getOptions();
025        javadoc.doFirst(new Action<Task>() {
026            @Override
027            public void execute(Task t) {
028                if (!t.getProject().getGradle().getStartParameter().isOffline())
029                    javadocOptions.setLinks(extension.javadocLinks);
030                javadocOptions.setLinkSource(extension.javadocLinkSource);
031                if (!extension.javadocGroups.isEmpty())
032                    javadocOptions.setGroups(extension.javadocGroups);
033                if (JavaVersion.current().isJava8Compatible())
034                    if (extension.javadocQuiet)
035                        javadocOptions.addStringOption("Xdoclint:none", "-quiet");
036            }
037        });
038    }
039}
