006package org.anarres.gradle.plugin.stdproject;
007
008import groovy.lang.GroovyObjectSupport;
009import java.net.URI;
010import java.util.HashMap;
011import java.util.Map;
012import javax.annotation.Nonnull;
013
014/**
015 *
016 * @author shevek
017 */
018public class License extends GroovyObjectSupport {
019
020    public static final Map<String, License> LICENSES = new HashMap<String, License>() {
021        private void add(@Nonnull String name, @Nonnull String title, @Nonnull String url) {
022            put(name, new License(name, title, URI.create(url)));
023        }
024
025        {
026            add("Apache-2.0", "Apache License, Version 2.0", "http://www.apache.org/licenses/LICENSE-2.0.html");
027            add("GPL-3.0", "GNU General Public License, Version 3.0", "http://www.gnu.org/licenses/gpl.html");
028            add("GPL-2.0", "GNU General Public License, Version 2.0", "http://www.gnu.org/licenses/old-licenses/gpl-2.0.html");
029            add("GPL-1.0", "GNU General Public License, Version 1.0", "http://www.gnu.org/licenses/old-licenses/gpl-1.0.html");
030            add("LGPL-3.0", "GNU Lesser General Public License, Version 3.0", "http://www.gnu.org/licenses/lgpl.html");
031            add("LGPL-2.1", "GNU Lesser General Public License, Version 2.1", "http://www.gnu.org/licenses/old-licenses/lgpl-2.1.html");
032            add("MIT", "MIT License", "http://opensource.org/licenses/MIT");
033        }
034    };
035    private String name;
036    private String description;
037    private URI uri;
038
039    public License() {
040    }
041
042    public License(@Nonnull String name, @Nonnull String description, @Nonnull URI uri) {
043        this.name = name;
044        this.description = description;
045        this.uri = uri;
046    }
047
048    public String getName() {
049        return name;
050    }
051
052    public void setName(@Nonnull String name) {
053        this.name = name;
054    }
055
056    public String getDescription() {
057        return description;
058    }
059
060    public void setDescription(@Nonnull String description) {
061        this.description = description;
062    }
063
064    public URI getUri() {
065        return uri;
066    }
067
068    public void setUri(@Nonnull URI uri) {
069        this.uri = uri;
070    }
071
072    public void setUri(@Nonnull String uri) {
073        setUri(URI.create(uri));
074    }
075
076}
