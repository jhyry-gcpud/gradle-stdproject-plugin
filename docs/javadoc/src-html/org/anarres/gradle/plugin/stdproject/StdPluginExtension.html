001package org.anarres.gradle.plugin.stdproject;

003import groovy.lang.GroovyObjectSupport;
004import java.util.HashMap;
005import java.util.LinkedHashSet;
006import java.util.Map;
007import java.util.Set;
008import javax.annotation.Nonnull;
009import org.gradle.api.Project;
010
011/**
012 * The standard plugin plugin extension.
013 *
014 * This allows configuring the standard plugin plugin using a
015 * <code>stdplugin { }</code> block.
016 *
017 * @author shevek
018 */
019public class StdPluginExtension extends GroovyObjectSupport {
020
021    // public String pluginName;
022    // public String pluginDescription;
023    public Map<String, String> implementations = new HashMap<String, String>();
024    // public final Set<String> pluginLicenses = new LinkedHashSet<String>();
025    public final Set<String> tags = new LinkedHashSet<String>();
026
027    public StdPluginExtension(@Nonnull Project project) {
028        // pluginName = project.getName();
029        // pluginLicenses.add("Apache-2.0");
030    }
031
032    public void implementation(@Nonnull String id, @Nonnull String implementationClass) {
033        implementations.put(id, implementationClass);
034    }
035
036    public void implementation(@Nonnull String implementationClass) {
037        String id = implementationClass.
038                replaceAll("\\.gradle\\.", ".").
039                replaceAll("\\.plugin\\.", ".").
040                substring(0, implementationClass.lastIndexOf('.'));
041        implementation(id, implementationClass);
042    }
043
044    public void tags(Object... tags) {
045        for (Object tag : tags)
046            this.tags.add(String.valueOf(tag));
047    }
048}
