006package org.anarres.gradle.plugin.stdproject;
007
008import java.io.File;
009import java.util.Collections;
010import java.util.Map;
011import java.util.concurrent.Callable;
012import org.gradle.api.Action;
013import org.gradle.api.Plugin;
014import org.gradle.api.Project;
015import org.gradle.api.Task;
016import org.gradle.api.plugins.JavaPluginConvention;
017import org.gradle.api.publish.maven.plugins.MavenPublishPlugin;
018import org.gradle.api.tasks.SourceSetContainer;
019import org.gradle.plugin.devel.plugins.JavaGradlePluginPlugin;
020
021/**
022 *
023 * @author shevek
024 */
025public class StdPluginPlugin implements Plugin<Project> {
026
027    @Override
028    public void apply(Project project) {
029        project.getPlugins().apply(StdModulePlugin.class);
030
031        final StdPluginExtension extension = project.getExtensions().create("stdplugin", StdPluginExtension.class, project);
032
033        // Gradle
034        project.getPlugins().apply(JavaGradlePluginPlugin.class);
035        project.getRepositories().add(project.getRepositories().jcenter());
036        // project.getDependencies().add(JavaPlugin.COMPILE_CONFIGURATION_NAME, project.getDependencies().gradleApi());
037        // project.getDependencies().add(JavaPlugin.TEST_COMPILE_CONFIGURATION_NAME, project.getDependencies().gradleTestKit());
038
039        // Maven
040        project.getPlugins().apply(MavenPublishPlugin.class);
041
042        // Bintray
043        /*
044         project.getPlugins().apply(BintrayPlugin.class);
045         BintrayExtension bintray = project.getExtensions().getByType(BintrayExtension.class);
046         String bintrayUsername = StdProjectPlugin.getExtraPropertyOrNull(project.getRootProject(), "bintrayUsername");
047         if (bintrayUsername != null)
048         bintray.setUser(bintrayUsername);
049         String bintrayApiKey = StdProjectPlugin.getExtraPropertyOrNull(project.getRootProject(), "bintrayApiKey");
050         if (bintrayApiKey != null)
051         bintray.setKey(bintrayApiKey);
052         bintray.getPkg().setRepo("gradle-plugins");
053         */
054        // Plugin
055        final File pluginDescriptorDir = new File(project.getBuildDir(), "generated-resources/gradle-plugin");
056        Task generatePluginDescriptors = project.getTasks().create("generatePluginDescriptors", GeneratePluginDescriptors.class, new Action<GeneratePluginDescriptors>() {
057            @Override
058            public void execute(GeneratePluginDescriptors t) {
059                // t.setGroup("Build");
060                t.setDescription("Generates gradle plugin descriptors.");
061                t.setDestinationDir(pluginDescriptorDir);
062                t.conventionMapping("pluginImplementations", new Callable<Map<String, String>>() {
063                    @Override
064                    public Map<String, String> call() throws Exception {
065                        return extension.implementations;
066                    }
067                });
068            }
069        });
070        SourceSetContainer sourceSets = project.getConvention().getPlugin(JavaPluginConvention.class).getSourceSets();
071        sourceSets.getByName("main").getOutput().dir(Collections.<String, Object>singletonMap("builtBy", generatePluginDescriptors), pluginDescriptorDir);
072    }
073
074}
